# Cloning

 ```
 git clone --recursive git@github.com:mlanca-c/Minitalk.git
 cd Minitalk
 make all
 ```
 
 # Compiling
 
 ```
 $ ./server
 PID: <PID_SERVER>
 
 (wait)
 ```

 ```
 $ ./client <PID_SERVER> <STRING>
 ```
